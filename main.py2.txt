import tkinter as tk
from tkinter import ttk, messagebox
import requests
import json
import os
from datetime import datetime

class GitHubUserFinder:
    def __init__(self, root):
        self.root = root
        self.root.title("GitHub User Finder")
        self.root.geometry("800x600")
        
        # Файл для хранения избранного
        self.favorites_file = "favorites.json"
        
        # Загружаем избранных пользователей
        self.favorites = self.load_favorites()
        
        # Создаем интерфейс
        self.create_widgets()
        
    def load_favorites(self):
        """Загрузка избранных пользователей из JSON"""
        if os.path.exists(self.favorites_file):
            try:
                with open(self.favorites_file, 'r', encoding='utf-8') as f:
                    return json.load(f)
            except:
                return []
        return []
    
    def save_favorites(self):
        """Сохранение избранных пользователей в JSON"""
        with open(self.favorites_file, 'w', encoding='utf-8') as f:
            json.dump(self.favorites, f, indent=2, ensure_ascii=False)
    
    def create_widgets(self):
        """Создание элементов интерфейса"""
        
        # Основной фрейм
        main_frame = ttk.Frame(self.root, padding="10")
        main_frame.grid(row=0, column=0, sticky=(tk.W, tk.E, tk.N, tk.S))
        
        # Настройка расширения
        self.root.columnconfigure(0, weight=1)
        self.root.rowconfigure(0, weight=1)
        main_frame.columnconfigure(1, weight=1)
        
        # Поиск
        ttk.Label(main_frame, text="Поиск пользователя GitHub:").grid(row=0, column=0, sticky=tk.W, pady=(0, 5))
        
        search_frame = ttk.Frame(main_frame)
        search_frame.grid(row=1, column=0, columnspan=2, sticky=(tk.W, tk.E), pady=(0, 10))
        search_frame.columnconfigure(0, weight=1)
        
        self.search_var = tk.StringVar()
        self.search_entry = ttk.Entry(search_frame, textvariable=self.search_var, width=50)
        self.search_entry.grid(row=0, column=0, padx=(0, 5), sticky=(tk.W, tk.E))
        
        self.search_entry.bind('<Return>', lambda e: self.search_user())
        
        search_btn = ttk.Button(search_frame, text="Поиск", command=self.search_user)
        search_btn.grid(row=0, column=1)
        
        # Панель с вкладками
        self.notebook = ttk.Notebook(main_frame)
        self.notebook.grid(row=2, column=0, columnspan=2, sticky=(tk.W, tk.E, tk.N, tk.S), pady=(0, 10))
        
        main_frame.rowconfigure(2, weight=1)
        
        # Вкладка результатов поиска
        self.results_frame = ttk.Frame(self.notebook)
        self.notebook.add(self.results_frame, text="Результаты поиска")
        
        # Вкладка избранного
        self.favorites_frame = ttk.Frame(self.notebook)
        self.notebook.add(self.favorites_frame, text="Избранное")
        
        # Создаем Treeview для результатов
        self.create_results_treeview()
        
        # Создаем Treeview для избранного
        self.create_favorites_treeview()
        
        # Кнопки управления
        btn_frame = ttk.Frame(main_frame)
        btn_frame.grid(row=3, column=0, columnspan=2, pady=(10, 0))
        
        self.add_to_fav_btn = ttk.Button(btn_frame, text="Добавить в избранное", 
                                        command=self.add_to_favorites, state='disabled')
        self.add_to_fav_btn.grid(row=0, column=0, padx=5)
        
        remove_btn = ttk.Button(btn_frame, text="Удалить из избранного", 
                               command=self.remove_from_favorites)
        remove_btn.grid(row=0, column=1, padx=5)
        
        refresh_btn = ttk.Button(btn_frame, text="Обновить избранное", 
                                command=self.refresh_favorites)
        refresh_btn.grid(row=0, column=2, padx=5)
        
        # Статус бар
        self.status_var = tk.StringVar()
        self.status_var.set("Готов к поиску")
        status_bar = ttk.Label(main_frame, textvariable=self.status_var, relief=tk.SUNKEN)
        status_bar.grid(row=4, column=0, columnspan=2, sticky=(tk.W, tk.E), pady=(10, 0))
    
    def create_results_treeview(self):
        """Создание таблицы для результатов поиска"""
        columns = ('Логин', 'Имя', 'Компания', 'Локация', 'Публичные репозитории', 'Подписчики')
        
        self.results_tree = ttk.Treeview(self.results_frame, columns=columns, show='headings')
        
        # Настройка колонок
        self.results_tree.heading('Логин', text='Логин')
        self.results_tree.heading('Имя', text='Имя')
        self.results_tree.heading('Компания', text='Компания')
        self.results_tree.heading('Локация', text='Локация')
        self.results_tree.heading('Публичные репозитории', text='Публичные репозитории')
        self.results_tree.heading('Подписчики', text='Подписчики')
        
        self.results_tree.column('Логин', width=150)
        self.results_tree.column('Имя', width=150)
        self.results_tree.column('Компания', width=150)
        self.results_tree.column('Локация', width=150)
        self.results_tree.column('Публичные репозитории', width=150)
        self.results_tree.column('Подписчики', width=100)
        
        # Скроллбар
        scrollbar = ttk.Scrollbar(self.results_frame, orient=tk.VERTICAL, command=self.results_tree.yview)
        self.results_tree.configure(yscrollcommand=scrollbar.set)
        
        # Размещение
        self.results_tree.grid(row=0, column=0, sticky=(tk.W, tk.E, tk.N, tk.S))
        scrollbar.grid(row=0, column=1, sticky=(tk.N, tk.S))
        
        self.results_frame.columnconfigure(0, weight=1)
        self.results_frame.rowconfigure(0, weight=1)
        
        # Привязка события выбора
        self.results_tree.bind('<<TreeviewSelect>>', self.on_result_select)
    
    def create_favorites_treeview(self):
        """Создание таблицы для избранного"""
        columns = ('Логин', 'Имя', 'Компания', 'Локация', 'Публичные репозитории', 'Подписчики', 'Дата добавления')
        
        self.favorites_tree = ttk.Treeview(self.favorites_frame, columns=columns, show='headings')
        
        # Настройка колонок
        self.favorites_tree.heading('Логин', text='Логин')
        self.favorites_tree.heading('Имя', text='Имя')
        self.favorites_tree.heading('Компания', text='Компания')
        self.favorites_tree.heading('Локация', text='Локация')
        self.favorites_tree.heading('Публичные репозитории', text='Публичные репозитории')
        self.favorites_tree.heading('Подписчики', text='Подписчики')
        self.favorites_tree.heading('Дата добавления', text='Дата добавления')
        
        self.favorites_tree.column('Логин', width=120)
        self.favorites_tree.column('Имя', width=120)
        self.favorites_tree.column('Компания', width=120)
        self.favorites_tree.column('Локация', width=120)
        self.favorites_tree.column('Публичные репозитории', width=120)
        self.favorites_tree.column('Подписчики', width=100)
        self.favorites_tree.column('Дата добавления', width=120)
        
        # Скроллбар
        scrollbar = ttk.Scrollbar(self.favorites_frame, orient=tk.VERTICAL, command=self.favorites_tree.yview)
        self.favorites_tree.configure(yscrollcommand=scrollbar.set)
        
        # Размещение
        self.favorites_tree.grid(row=0, column=0, sticky=(tk.W, tk.E, tk.N, tk.S))
        scrollbar.grid(row=0, column=1, sticky=(tk.N, tk.S))
        
        self.favorites_frame.columnconfigure(0, weight=1)
        self.favorites_frame.rowconfigure(0, weight=1)
        
        # Заполняем избранное
        self.populate_favorites_tree()
    
    def search_user(self):
        """Поиск пользователя через GitHub API"""
        username = self.search_var.get().strip()
        
        # Проверка ввода
        if not username:
            messagebox.showwarning("Ошибка", "Поле поиска не может быть пустым!")
            return
        
        self.status_var.set(f"Поиск пользователя {username}...")
        self.root.update()
        
        try:
            # Запрос к GitHub API
            url = f"https://api.github.com/search/users?q={username}&per_page=20"
            response = requests.get(url)
            response.raise_for_status()
            
            data = response.json()
            
            # Очищаем предыдущие результаты
            for item in self.results_tree.get_children():
                self.results_tree.delete(item)
            
            # Если нет результатов
            if data['total_count'] == 0:
                self.status_var.set(f"Пользователи по запросу '{username}' не найдены")
                messagebox.showinfo("Результат", "Пользователи не найдены")
                return
            
            # Получаем детальную информацию для каждого пользователя
            users = data['items']
            detailed_users = []
            
            for user in users:
                try:
                    user_url = f"https://api.github.com/users/{user['login']}"
                    user_response = requests.get(user_url)
                    if user_response.status_code == 200:
                        user_data = user_response.json()
                        detailed_users.append(user_data)
                except:
                    continue
            
            # Отображаем результаты
            for user in detailed_users:
                self.results_tree.insert('', 'end', values=(
                    user.get('login', ''),
                    user.get('name', ''),
                    user.get('company', ''),
                    user.get('location', ''),
                    user.get('public_repos', 0),
                    user.get('followers', 0)
                ))
            
            self.status_var.set(f"Найдено {len(detailed_users)} пользователей")
            
        except requests.exceptions.RequestException as e:
            messagebox.showerror("Ошибка", f"Ошибка при подключении к API: {str(e)}")
            self.status_var.set("Ошибка подключения")
        except Exception as e:
            messagebox.showerror("Ошибка", f"Произошла ошибка: {str(e)}")
            self.status_var.set("Ошибка")
    
    def on_result_select(self, event):
        """Обработка выбора пользователя в результатах"""
        selection = self.results_tree.selection()
        if selection:
            self.add_to_fav_btn.config(state='normal')
    
    def add_to_favorites(self):
        """Добавление выбранного пользователя в избранное"""
        selection = self.results_tree.selection()
        if not selection:
            messagebox.showwarning("Ошибка", "Выберите пользователя для добавления в избранное")
            return
        
        # Получаем данные пользователя
        user_data = self.results_tree.item(selection[0])['values']
        
        # Проверяем, не добавлен ли уже пользователь
        login = user_data[0]
        for fav in self.favorites:
            if fav.get('login') == login:
                messagebox.showinfo("Информация", "Пользователь уже в избранном")
                return
        
        # Создаем запись для избранного
        favorite_user = {
            'login': login,
            'name': user_data[1],
            'company': user_data[2],
            'location': user_data[3],
            'public_repos': user_data[4],
            'followers': user_data[5],
            'added_date': datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        }
        
        # Добавляем в список избранного
        self.favorites.append(favorite_user)
        
        # Сохраняем в файл
        self.save_favorites()
        
        # Обновляем отображение избранного
        self.populate_favorites_tree()
        
        messagebox.showinfo("Успех", f"Пользователь {login} добавлен в избранное")
        self.status_var.set(f"Добавлен в избранное: {login}")
    
    def populate_favorites_tree(self):
        """Заполнение таблицы избранных пользователей"""
        # Очищаем таблицу
        for item in self.favorites_tree.get_children():
            self.favorites_tree.delete(item)
        
        # Добавляем пользователей
        for user in self.favorites:
            self.favorites_tree.insert('', 'end', values=(
                user.get('login', ''),
                user.get('name', ''),
                user.get('company', ''),
                user.get('location', ''),
                user.get('public_repos', 0),
                user.get('followers', 0),
                user.get('added_date', '')
            ))
    
    def remove_from_favorites(self):
        """Удаление выбранного пользователя из избранного"""
        selection = self.favorites_tree.selection()
        if not selection:
            messagebox.showwarning("Ошибка", "Выберите пользователя для удаления из избранного")
            return
        
        # Получаем логин пользователя
        user_data = self.favorites_tree.item(selection[0])['values']
        login = user_data[0]
        
        # Удаляем из списка
        self.favorites = [user for user in self.favorites if user.get('login') != login]
        
        # Сохраняем изменения
        self.save_favorites()
        
        # Обновляем отображение
        self.populate_favorites_tree()
        
        messagebox.showinfo("Успех", f"Пользователь {login} удален из избранного")
        self.status_var.set(f"Удален из избранного: {login}")
    
    def refresh_favorites(self):
        """Обновление списка избранного"""
        self.populate_favorites_tree()
        self.status_var.set("Список избранного обновлен")

def main():
    root = tk.Tk()
    app = GitHubUserFinder(root)
    root.mainloop()

if __name__ == "__main__":
    main()
