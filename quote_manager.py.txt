import json
import random
from datetime import datetime
from typing import List, Dict, Optional

class QuoteManager:
    """Класс для управления цитатами"""
    
    def __init__(self, quotes_file: str = "quotes.json", history_file: str = "history.json"):
        self.quotes_file = quotes_file
        self.history_file = history_file
        self.all_quotes = self.load_quotes()
        self.history = self.load_history()
        
    def load_quotes(self) -> List[Dict]:
        """Загрузка цитат из файла"""
        try:
            with open(self.quotes_file, 'r', encoding='utf-8') as f:
                return json.load(f)
        except FileNotFoundError:
            # Создаем файл с примерами цитат, если его нет
            default_quotes = [
                {
                    "id": 1,
                    "text": "Жизнь — это то, что происходит с тобой, пока ты строишь другие планы.",
                    "author": "Джон Леннон",
                    "topic": "Жизнь",
                    "added_date": "2024-01-01"
                },
                {
                    "id": 2,
                    "text": "Единственный способ сделать великую работу — любить то, что ты делаешь.",
                    "author": "Стив Джобс",
                    "topic": "Работа",
                    "added_date": "2024-01-01"
                }
            ]
            self.save_quotes(default_quotes)
            return default_quotes
    
    def save_quotes(self, quotes: List[Dict]) -> None:
        """Сохранение цитат в файл"""
        with open(self.quotes_file, 'w', encoding='utf-8') as f:
            json.dump(quotes, f, ensure_ascii=False, indent=2)
    
    def load_history(self) -> List[Dict]:
        """Загрузка истории из файла"""
        try:
            with open(self.history_file, 'r', encoding='utf-8') as f:
                return json.load(f)
        except FileNotFoundError:
            return []
    
    def save_history(self, history: List[Dict]) -> None:
        """Сохранение истории в файл"""
        with open(self.history_file, 'w', encoding='utf-8') as f:
            json.dump(history, f, ensure_ascii=False, indent=2)
    
    def get_random_quote(self, topic: str = None) -> Optional[Dict]:
        """Получение случайной цитаты"""
        if not self.all_quotes:
            return None
            
        if topic and topic != "Все темы":
            filtered_quotes = [q for q in self.all_quotes if q["topic"] == topic]
            if not filtered_quotes:
                return None
            return random.choice(filtered_quotes)
        
        return random.choice(self.all_quotes)
    
    def add_quote(self, text: str, author: str, topic: str) -> Dict:
        """Добавление новой цитаты"""
        new_id = max([q["id"] for q in self.all_quotes], default=0) + 1
        new_quote = {
            "id": new_id,
            "text": text,
            "author": author,
            "topic": topic,
            "added_date": datetime.now().strftime("%Y-%m-%d")
        }
        
        self.all_quotes.append(new_quote)
        self.save_quotes(self.all_quotes)
        return new_quote
    
    def add_to_history(self, quote: Dict) -> None:
        """Добавление цитаты в историю"""
        history_entry = {
            "quote_id": quote["id"],
            "text": quote["text"],
            "author": quote["author"],
            "topic": quote["topic"],
            "viewed_date
                        "viewed_date": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        }
        
        self.history.append(history_entry)
        self.save_history(self.history)
    
    def get_history(self, limit: int = None) -> List[Dict]:
        """Получение истории просмотров"""
        if limit and len(self.history) > limit:
            return self.history[-limit:]
        return self.history
    
    def get_all_topics(self) -> List[str]:
        """Получение всех тем"""
        topics = list(set(q["topic"] for q in self.all_quotes))
        return sorted(topics)
    
    def search_quotes(self, search_text: str) -> List[Dict]:
        """Поиск цитат по тексту или автору"""
        search_text = search_text.lower()
        results = []
        
        for quote in self.all_quotes:
            if (search_text in quote["text"].lower() or 
                search_text in quote["author"].lower()):
                results.append(quote)
        
        return results
    
    def get_quote_by_id(self, quote_id: int) -> Optional[Dict]:
        """Получение цитаты по ID"""
        for quote in self.all_quotes:
            if quote["id"] == quote_id:
                return quote
        return None
    
    def delete_quote(self, quote_id: int) -> bool:
        """Удаление цитаты"""
        for i, quote in enumerate(self.all_quotes):
            if quote["id"] == quote_id:
                del self.all_quotes[i]
                self.save_quotes(self.all_quotes)
                return True
        return False
