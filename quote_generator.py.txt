    def generate_quote(self):
        """Генерация случайной цитаты"""
        if not self.all_quotes:
            messagebox.showwarning("Нет цитат", "Добавьте хотя бы одну цитату!")
            return
            
        # Фильтрация по теме, если выбрана
        selected_topic = self.topic_var.get()
        if selected_topic and selected_topic != "Все темы":
            filtered_quotes = [q for q in self.all_quotes if q["topic"] == selected_topic]
            if not filtered_quotes:
                messagebox.showinfo("Нет цитат", f"Нет цитат по теме '{selected_topic}'")
                return
